import math

def calculate_entropy(dataset):
    # Calculate the number of instances in the dataset
    num_instances = len(dataset)
    
    label_counts = {}
    for item in dataset:
        label = item[-1]
        if label not in label_counts:
            label_counts[label] = 0
        label_counts[label] += 1

    entropy = 0
    for label, count in label_counts.items():
        p = count / num_instances
        entropy -= p * math.log(p, 2)

    return entropy

def information_gain(data, split_attribute, label_attribute):
    # Calculate the entropy of the dataset
    data_entropy = calculate_entropy(data)

    # Create subsets based on the given attribute
    subsets = {}
    for item in data:
        attribute_value = item[split_attribute]
        if attribute_value not in subsets:
            subsets[attribute_value] = []
        subsets[attribute_value].append(item)

    # Calculate the weight of each subset
    weight = [float(len(subset)) / len(data) for subset in subsets.values()]

    # Calculate the entropy of each subset
    subset_entropy = [calculate_entropy(subset) for subset in subsets.values()]

    # Calculate the sum of the product of each subset entropy and weight
    weighted_entropy = sum(w * e for w, e in zip(weight, subset_entropy))

    # Return the information gain
    return data_entropy - weighted_entropy

def split_data(data, attribute):
    subsets = {}

        if attribute_value not in subsets:
            subsets[attribute_value] = []
        subsets[attribute_value].append(item)

    return [subset for attribute_value, subset in subsets.items()]

def build_tree(data, features, labels, depth=0):
    if len(data) == 0:
        return {"": {"": {}}}

    if labels.unique().size == 1:
        return {"": {"": labels.iloc[0]}}

    best_feature = None
    max_info_gain = -1

    for feature in features:
        info_gain = information_gain(data, feature, labels)
        if info_gain > max_info_gain:
            max_info_gain = info_gain
            best_feature = feature

    if max_info_gain == 0:
        return {"": {"": labels.value_counts().idxmax()}}

    subsets = split_data(data, best_feature)
    sub_features = features.copy()
    sub_features.remove(best_feature)

    tree = {"": {}}
    for value, subset in subsets.items():
        tree[""][value] = build_tree(subset, sub_features, labels, depth=depth + 1)

    return tree

def predict(tree, row):

    for key, value in tree.items():
        if isinstance(value, dict):
            prediction = predict(value, row)
